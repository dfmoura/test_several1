select 
status, status_descricao, count(*) qtd
from (
select w.*,
case
when status = 'ENC' then 'PEDIDO ENCERRADO'
when status = 'ME' then 'MERCADORIA ENTREGUE'
when status = 'MA' then 'MERCDORIA A CAMINHO'
when status = 'PE' then 'PREVISAO ENTREGA'
when status = 'ANF' then 'AGUARDANDO NF'
when status = 'NFE' then 'NOTA FISCAL EMITIDA'
when status = 'PS' then 'PEDIDO EM SEPARACAO'
when status = 'R' then 'REJEITADO'
when status = 'A' then 'APROVADO'
when status = 'EE' then 'EM ANALISE'
when status = 'C' then 'CANCELADO'
when status = 'PFP' then 'PEDIDO FATURADO PARCIAL'
END AS STATUS_DESCRICAO

FROM AD_VGFSTATUSA2W W
)
GROUP BY STATUS,STATUS_DESCRICAO


SELECT
A2W.NUNOTA,
A2W.STATUS,
A2W.COMENTARIO,
A2W.PERC_FATURADO,
CAB.DTNEG,
CAB.CODVEND,
VEN.APELIDO,
CAB.CODPARC,
PAR.NOMEPARC,
CAB.VLRNOTA



FROM AD_VGFSTATUSA2W A2W
INNER JOIN TGFCAB CAB ON A2W.NUNOTA = CAB.NUNOTA
INNER JOIN TGFPAR PAR ON CAB.CODPARC = PAR.CODPARC
INNER JOIN TGFVEN VEN ON CAB.CODVEND = VEN.CODVEND
WHERE A2W.STATUS IN (/*'EE','A','PS','ANG','PFP'*/'NFE')
--'A'
--'ENC'
