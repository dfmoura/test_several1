    SELECT
        CASE WHEN VOA.DIVIDEMULTIPLICA = 'M' THEN ITE.QTDNEG / VOA.QUANTIDADE
             ELSE ITE.QTDNEG * VOA.QUANTIDADE END AS QTDNEG2
    --INTO V_QTDNEG
    FROM TGFITE ITE
    INNER JOIN TGFPRO PRO ON PRO.CODPROD = ITE.CODPROD
    LEFT JOIN TGFVOA VOA ON VOA.CODPROD = PRO.CODPROD AND ITE.CODVOL = VOA.CODVOL
    WHERE ITE.NUNOTA = 77569--P_NUNOTA 
    AND ITE.CODPROD = 168--P_CODPROD;
